技术原理

  1. 页面加载时创建虚拟下载链接
  2. 使用 application/octet-stream MIME 类型
  3. 触发微信内置下载功能
  4. 微信安卓版不支持此下载类型
  5. 自动弹出"在浏览器中打开"提示
  6. 用户点击"允许"后跳转至浏览器

修复的核心代码

以下为解决自动触发问题的关键代码：

// 更可靠的微信环境检测
function isWechatBrowser() {
  const ua = navigator.userAgent.toLowerCase();
  return ua.indexOf('micromessenger') !== -1;
}

// 修复的触发函数
function triggerWechatPopup() {
  try {
    // 创建虚拟下载链接
    const downloadLink = document.createElement('a');
    
    // 使用无法识别的文件类型
    downloadLink.href = 'data:application/octet-stream;base64,' + btoa('wechat-jumper');
    downloadLink.download = 'open_in_browser.bin';
    
    // 添加到DOM并触发点击
    document.body.appendChild(downloadLink);
    const event = new MouseEvent('click', {
      view: window,
      bubbles: true,
      cancelable: true
    });
    downloadLink.dispatchEvent(event);
    
    // 清理DOM
    setTimeout(() => {
      document.body.removeChild(downloadLink);
    }, 1000);
    
    return true;
  } catch (e) {
    console.error('触发失败:', e);
    return false;
  }
}

// 页面加载后立即触发
window.addEventListener('load', function() {
  if (isWechatBrowser()) {
    // 添加延迟确保微信环境完全加载
    setTimeout(() => {
      const success = triggerWechatPopup();
      if (success) {
        updateStatus('triggerStatus', '已触发', 'ok');
      } else {
        updateStatus('triggerStatus', '触发失败', 'warn');
      }
    }, 1500);
  }
});
